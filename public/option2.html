Express Checkout Demo

Separate Charges & Transfers with on_behalf_of

← Back Option 1: Client-side Confirm Option 2: 2-Step ConfirmationServer-side
Sample Product
$25.00
Flow (2-Step Confirmation):
1. Frontend creates Express Checkout Element (no PaymentIntent yet)
2. Customer interacts with Apple Pay / Google Pay
3. Frontend calls stripe.createConfirmationToken()
4. Backend receives token, creates PaymentIntent with confirm: true
5. Backend creates a separate Transfer to connected account
6. Frontend shows result
Frontend Code (JavaScript) 
const AMOUNT = 2500;

async function initialize() {
  const { publishableKey, connectedAccountId } =
    await fetch("/api/config").then(r => r.json());
  const stripe = Stripe(publishableKey);

  // No PaymentIntent created upfront — deferred mode
  const elements = stripe.elements({
    mode: "payment",
    amount: AMOUNT,
    currency: "usd",
    onBehalfOf: connectedAccountId,
  });

  const ece = elements.create("expressCheckout", {
    paymentMethods: { applePay: "always", googlePay: "always" },
  });
  ece.mount("#express-checkout-element");

  ece.on("confirm", async () => {
    // Step 1: Create ConfirmationToken client-side
    const { confirmationToken } =
      await stripe.createConfirmationToken({ elements });

    // Step 2: Send token to backend for server-side confirmation
    const result = await fetch("/api/confirm-payment", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        confirmationTokenId: confirmationToken.id,
        amount: AMOUNT,
        currency: "usd",
      }),
    }).then(r => r.json());

    // Handle 3DS if needed
    if (result.status === "requires_action") {
      await stripe.handleNextAction({
        clientSecret: result.clientSecret,
      });
    }
  });
}
Backend Code (Node.js / Express) 
const stripe = require("stripe")(process.env.STRIPE_SECRET_KEY);

// POST /api/confirm-payment
app.post("/api/confirm-payment", async (req, res) => {
  const { confirmationTokenId, amount, currency } = req.body;

  // Create and confirm PaymentIntent in one API call
  const paymentIntent = await stripe.paymentIntents.create({
    amount,
    currency,
    on_behalf_of: process.env.STRIPE_CONNECTED_ACCOUNT_ID,
    confirmation_token: confirmationTokenId,
    confirm: true,
    automatic_payment_methods: { enabled: true },
    return_url: "https://your-site.com/option2.html",
  });

  if (paymentIntent.status === "requires_action") {
    return res.json({
      status: "requires_action",
      clientSecret: paymentIntent.client_secret,
    });
  }

  if (paymentIntent.status === "succeeded") {
    // Create transfer immediately after confirmation
    const transfer = await stripe.transfers.create({
      amount,
      currency,
      destination: process.env.STRIPE_CONNECTED_ACCOUNT_ID,
      source_transaction: paymentIntent.latest_charge,
    });

    res.json({
      status: "succeeded",
      transferId: transfer.id,
    });
  }
});
