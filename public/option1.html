Express Checkout Demo

Separate Charges & Transfers with on_behalf_of

← Back Option 1: Client-side Confirm Option 2: 2-Step Confirmation
Sample Product
$25.00
Flow:
1. Platform creates PaymentIntent with on_behalf_of
2. Customer pays via Express Checkout (Apple Pay / Google Pay)
3. stripe.confirmPayment() completes the charge client-side
4. Platform creates a separate Transfer to connected account
Frontend Code (JavaScript) 
const AMOUNT = 2500;

async function initialize() {
  // Fetch config from backend
  const { publishableKey, connectedAccountId } =
    await fetch("/api/config").then(r => r.json());
  const stripe = Stripe(publishableKey);

  // Create PaymentIntent upfront
  const { clientSecret, paymentIntentId } = await fetch("/api/create-payment-intent", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ amount: AMOUNT, currency: "usd" }),
  }).then(r => r.json());

  // Initialize Elements
  const elements = stripe.elements({
    mode: "payment",
    amount: AMOUNT,
    currency: "usd",
    onBehalfOf: connectedAccountId,
  });

  // Mount Express Checkout Element
  const ece = elements.create("expressCheckout", {
    paymentMethods: { applePay: "always", googlePay: "always" },
  });
  ece.mount("#express-checkout-element");

  // Confirm payment client-side
  ece.on("confirm", async () => {
    const { error } = await stripe.confirmPayment({
      elements, clientSecret,
      confirmParams: { return_url: window.location.href },
      redirect: "if_required",
    });
    if (!error) await createTransfer(paymentIntentId);
  });
}

async function createTransfer(paymentIntentId) {
  await fetch("/api/create-transfer", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ amount: AMOUNT, currency: "usd", paymentIntentId }),
  });
}
Backend Code (Node.js / Express) 
const stripe = require("stripe")(process.env.STRIPE_SECRET_KEY);

// POST /api/create-payment-intent
app.post("/api/create-payment-intent", async (req, res) => {
  const { amount, currency } = req.body;

  const paymentIntent = await stripe.paymentIntents.create({
    amount,
    currency,
    on_behalf_of: process.env.STRIPE_CONNECTED_ACCOUNT_ID,
    automatic_payment_methods: { enabled: true },
  });

  res.json({
    clientSecret: paymentIntent.client_secret,
    paymentIntentId: paymentIntent.id,
  });
});

// POST /api/create-transfer
app.post("/api/create-transfer", async (req, res) => {
  const { amount, currency, paymentIntentId } = req.body;

  // Retrieve the charge ID from the PaymentIntent
  const pi = await stripe.paymentIntents.retrieve(paymentIntentId);

  const transfer = await stripe.transfers.create({
    amount,
    currency,
    destination: process.env.STRIPE_CONNECTED_ACCOUNT_ID,
    source_transaction: pi.latest_charge,
  });

  res.json({ transferId: transfer.id });
});
